# node-minify

## Installation

```bash
npm install @node-minify/core @node-minify/uglify-js
```

## Usage

```js
const minify = require('@node-minify/core');
const uglifyJS = require('@node-minify/uglify-js');

minify({
  compressor: uglifyJS,
  input: 'foo.js',
  output: 'bar.js',
  callback: function (err, min) {}
});
```
